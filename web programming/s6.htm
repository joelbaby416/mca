JavaScript String and Math Objects

string object

String Length

	 				
	 					
	 	var str = "Hello, World!";
	 	var length = str.length;
	 	console.log("Length of the string: " + length);

String Concatenation

	 					
	 	
	 	var str1 = "Hello";
	 	
	 	var str2 = "World";
	 	
	 	var result = str1.concat(",", str2, "!");
	 	
	  console.log("Concatenated string: " + result);
